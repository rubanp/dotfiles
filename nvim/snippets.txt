Custom Element
==============
import minireset from './minireset';
class CustomElem extends HTMLElement {
  constructor() {
    super();
    const shadow = this.attachShadow({ mode: 'open' });
    shadow.innerHTML = `
    <style>
        h1 {
            color: blue;
        }
    </style>
    <h1>Hello World</h1>
    <ul>
        <li>How you doing?</li>
    </ul>
    `;
    shadow.prependChild(minireset);
  }
}
export default CustomElem;
